CREATE TABLE secoes(
cd int( 10 ) unsigned NOT NULL AUTO_INCREMENT ,
titulo varchar( 255 ) NOT NULL default '',
texto LONGTEXT NOT NULL ,
nomedesecao int( 10 ) unsigned NOT NULL default '0',
PRIMARY KEY ( cd ) 
);

CREATE TABLE `nomedesecao` (
`cd` int( 10 ) unsigned NOT NULL AUTO_INCREMENT ,
`nome` varchar( 255 ) NOT NULL default '',
`pgseparadas` char( 1 ) NOT NULL default 'n',
PRIMARY KEY ( `cd` ) 
);
